NAME
    Lingua::Align::Corpus - Perl extension for reading a tokenized plain
    text corpus, 1 sentence per line

SYNOPSIS
      use Lingua::Align::Corpus;

      my $corpus = new Lingua::Align::Corpus(-file => $corpusfile);

      my @words=();
      while ($corpus->next_sentence(\@words)){
        print "\n",$corpus->current_id,"> ";
        print join(' ',@words);
      }
